je_com_supranational_organization = {
	group = je_group_com_supranational_organizations
	icon = "gfx/interface/icons/invention_icons/pan_nationalism.dds"
	weight = 100
	
	widget = {
		gui = "gui/com_journal_injects/interorg_widgets.gui"
		name = "widget_com_supranational_organization"
		container = "custom_widget_container_je_icon"
	}
	
	scripted_button = sb_international_organizations_test_1
	scripted_button = sb_international_organizations_test_2
	scripted_button = sb_international_organizations_test_3
	
	should_be_involved = {
		country_rank >= rank_value:great_power
	}
	
	should_show_when_not_involved = {
		country_rank < rank_value:great_power
	}
	
	immediate = {
		scope:journal_entry = {
			c:GBR = {
				save_scope_as = interorg_country_scope
				ruler = {
					save_scope_as = interorg_character_scope
				}
			}
			set_variable = {
				name = com_interorg_leader_var
				value = scope:interorg_country_scope
			}
			set_variable = {
				name = com_interorg_chair_var
				value = scope:interorg_character_scope
			}
			com_set_international_organization_headquaters_effect = {
				state_region = s:STATE_FLANDERS
			}
			com_hide_all_journal_elements_for_interorg_effect = yes
		}
		com_activate_interorg_sidebar = yes
	}
	
	complete = {
		always = no
	}
	
	on_complete = {
		custom_tooltip = com_debug.a
	}
}
