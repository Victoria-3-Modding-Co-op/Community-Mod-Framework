je_com_supranational_organization = {
	group = je_group_com_supranational_organizations
	icon = "gfx/interface/icons/invention_icons/multilateral_alliances.dds"
	weight = 100
	
	widget = {
		gui = "gui/journal_entry_widgets/com_interorg_widgets.gui"
		name = "widget_com_supranational_organization_base"
		container = "custom_widget_container_je_icon"
	}
	
	scripted_button = sb_international_organizations_test_1
	scripted_button = sb_international_organizations_test_2
	scripted_button = sb_international_organizations_test_3
	
	should_be_involved = {
		country_rank >= rank_value:great_power
	}
	
	should_show_when_not_involved = {
		country_rank < rank_value:great_power
	}
	
	immediate = {
		scope:journal_entry = {
			com_build_international_organization_list_effect = {
				type = je_com_supranational_organization
			}
			com_update_international_ogranization_members_value_effect = {
				type = je_com_supranational_organization
			}
			com_set_international_organization_headquaters_effect = {
				state_region = s:STATE_FLANDERS
			}
			set_variable = com_hide_journal_reason
			set_variable = com_hide_involved_countries
			set_variable = com_hide_complete_triggers
			set_variable = com_hide_fail_triggers
		}
		com_activate_international_organization_sidebar = yes
	}
	
	complete = {
		always = no
	}
	
	on_complete = {
		custom_tooltip = com_debug.a
	}
	
	on_weekly_pulse = {
		effect = {
			scope:journal_entry = {
				com_build_international_organization_list_effect = {
					type = je_com_supranational_organization
				}
				com_update_international_ogranization_members_value_effect = {
					type = je_com_supranational_organization
				}
			}
		}
	}
}
