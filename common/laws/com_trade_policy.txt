REPLACE_OR_CREATE:law_isolationism = {
	group = lawgroup_trade_policy

	icon = "gfx/interface/icons/law_icons/isolationism.dds"

	progressiveness = -50

	disallowing_laws = {
		law_laissez_faire
	}

	is_visible = {
		OR = {
			AND = {
				com_law_total_seperation_trigger = yes
				NOR = {
					AND = {
						any_primary_culture = {
							has_discrimination_trait = heritage_han
						}
						any_scope_state = {
							state_region = s:STATE_GUANGDONG
							is_largest_state_in_region = yes
						}
					}
					AND = {
						any_primary_culture = {
							cu:japanese = this
						}
						s:STATE_KYUSHU ?= {
							any_scope_state = {
								owner ?= ROOT
								is_largest_state_in_region = yes
							}
						}
					}
				}
			}
			com_law_total_seperation_alternative_trigger = yes
		}
	}

	modifier = {
		country_authority_mult = 0.5
		state_tax_capacity_mult = 0.25
		country_tech_spread_mult = -0.15

		country_leverage_resistance_mult = 1.0
		country_support_separatism_resistance_mult = 0.5

		country_disallow_trade_bool = yes
	}

	ai_will_do = {
		OR = {
			has_journal_entry = je_gyojo_shinwon
			ruler = {
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_vanguardist
				}
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}

# Special isolationism variant for Japan, allows TCs to be built only in Dejima
REPLACE_OR_CREATE:law_sakoku = {
	group = lawgroup_trade_policy
	parent = law_isolationism # Inherits ideological stances from parent

	icon = "gfx/interface/icons/law_icons/sakoku.dds"

	progressiveness = -50

	requires_law_or = {
		law_traditionalism
	}

	is_visible = {
		OR = {
			AND = {
				com_law_sakoku_trigger = yes
				country_has_primary_culture = cu:japanese
				any_scope_state = {
					state_region = s:STATE_KYUSHU
					is_largest_state_in_region = yes
				}
			}
			com_law_sakoku_alternative_trigger = yes
		}
	}

	modifier = {
		country_authority_mult = 0.25
		country_influence_mult = -0.50
		country_tech_spread_mult = -0.20
		state_trade_capacity_mult = -0.75
		state_tariff_import_add = 0.50
		state_tariff_export_add = 0.50
		state_subvention_export_add = 0.20
		state_subvention_import_add = 0.20
		country_leverage_resistance_mult = 0.75

		country_disallow_trade_outside_kyushu_bool = yes
		state_trade_center_max_limit_add = 5
		
		country_disallow_aggressive_plays_bool = yes
	}

	ai_will_do = {
		OR = {
			has_journal_entry = je_sakoku
			ruler = {
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_vanguardist
				}
			}
		}
	}

	on_deactivate = {
		if = {
			limit = {
				has_law = law_type:law_closed_borders
			}
			custom_tooltip = internal_controls_relaxed_tt
			activate_law = law_type:law_migration_controls
		}
		custom_tooltip = {
			text = landowners_stop_supporting_sakoku_tt
			ig:ig_landowners ?= {
				remove_ideology = ideology_japan_hierarchic
				add_ideology = ideology_japan_hierarchic_2
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0

		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
}
