com_hide_international_organization_journal_entry_group = {
	add_to_global_variable_list = {
		name = com_international_organization_journal_groups
		target = flag:$name$
	}
	add_to_global_variable_list = {
		name = com_hidden_journal_groups
		target = flag:$name$
	}
}

com_activate_international_organization_sidebar = {
	set_global_variable = com_international_ogranization_var
}

com_set_international_organization_headquaters_effect = {
	set_variable = {
		name = com_organization_headquarters_var
		value = $state_region$
	}
}

com_update_international_ogranization_members_value_effect = {
	set_variable = {
		name = com_involved_countries_country_var
		value = {
			every_country = {
				limit = {
					is_involved_in_journal_entry = $type$
				}
				add = 1
			}
			min = 1
		}
	}
}

com_build_international_organization_list_effect = {
	clear_variable_list = com_involved_countries_display_list
	save_scope_as = com_journal_entry_scope
	ordered_country = {
		limit = {
			is_involved_in_journal_entry = $type$
		}
		order_by = prestige
		min = 1
		max = 10
		check_range_bounds = no # If you don't want an error logged if the list is smaller than the min/max
		scope:com_journal_entry_scope = {
			add_to_variable_list = {
				name = com_involved_countries_display_list
				target = prev
			}
		}
	}
}
