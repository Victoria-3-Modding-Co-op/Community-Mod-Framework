com_hide_interorg_journal_entry_group = {
	add_to_global_variable_list = {
		name = com_international_organization_journal_groups
		target = flag:$name$
	}
	add_to_global_variable_list = {
		name = com_hidden_journal_groups
		target = flag:$name$
	}
	hidden_effect = {
		com_activate_interorg_sidebar = yes
	}
}

com_add_interorg_journal_entry = {
	custom_tooltip = {
		text = com_add_interorg_journal_entry
		set_local_variable = {
			name = com_flag_loc
			flag:$type$
		}
		add_contextless_journal_entry = $type$
	}
	hidden_effect = {
		com_activate_interorg_sidebar = yes
	}
}

com_activate_interorg_sidebar = {
	## Create International Organization List
	com_hide_interorg_journal_entry_group = {
		name = je_group_com_supranational_organizations
	}
	com_hide_interorg_journal_entry_group = {
		name = je_group_com_regional_organizations
	}
	com_hide_interorg_journal_entry_group = {
		name = je_group_com_religious_organizations
	}
	com_hide_interorg_journal_entry_group = {
		name = je_group_com_non_governmental_organizations
	}
	com_hide_interorg_journal_entry_group = {
		name = je_group_com_sporting_organizations
	}
}

com_set_international_organization_headquaters_effect = {
	set_variable = {
		name = com_interorg_headquarters_var
		value = $state_region$
	}
}

com_update_interorg_members_value_effect = {
	set_variable = {
		name = com_interorg_involved_countries_var
		value = {
			every_country = {
				limit = {
					is_involved_in_journal_entry = $type$
				}
				add = 1
			}
			min = 1
		}
	}
}

com_build_interorg_list_effect = {
	clear_variable_list = com_interorg_involved_countries_list
	save_scope_as = com_journal_entry_scope
	ordered_country = {
		limit = {
			is_involved_in_journal_entry = $type$
		}
		order_by = prestige
		min = 1
		max = 10
		check_range_bounds = no # If you don't want an error logged if the list is smaller than the min/max
		scope:com_journal_entry_scope = {
			add_to_variable_list = {
				name = com_interorg_involved_countries_list
				target = prev
			}
		}
	}
}
