com_add_journal_group_to_interorg_list = {
	# This only adds the journal group to the interorg list for the scripted widget, they also need to be hidden using com_hide_journal_entry or else they will appear in the journal
	add_to_global_variable_list = {
		name = com_international_organization_journal_groups
		target = flag:$name$
	}
}

com_add_journal_group_to_situations_list = {
	# This adds the journal group to the situations list; this is for outliner functionality
	add_to_global_variable_list = {
		name = com_international_situation_journal_groups
		target = flag:$name$
	}
}

com_add_journal_group_to_interorg_list_and_hide_from_journal = {
	com_add_journal_group_to_interorg_list = {
		name = $name$
	}
	com_hide_journal_entry_group = {
		name = $name$
	}
}

com_add_interorg_journal_entry = {
	custom_tooltip = {
		text = com_add_interorg_journal_entry
		set_local_variable = {
			name = com_flag_loc
			flag:$type$
		}
		add_contextless_journal_entry = $type$
	}
}

com_activate_interorg_sidebar = {
	set_global_variable = com_international_organization_panel_active
	# Already set as IO, hide when using sidebar
	com_hide_journal_entry_group = {
		name = je_group_com_supranational_organizations
	}
	com_hide_journal_entry_group = {
		name = je_group_com_regional_organizations
	}
	com_hide_journal_entry_group = {
		name = je_group_com_religious_organizations
	}
	com_hide_journal_entry_group = {
		name = je_group_com_non_governmental_organizations
	}
	com_hide_journal_entry_group = {
		name = je_group_com_sporting_organizations
	}
}
