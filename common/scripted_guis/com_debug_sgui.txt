com_toggle_debug = {
	scope = none

	effect = {
		if = {
			limit = {
				NOT = {
					has_global_variable = com_debug
				}
			}
			set_global_variable = com_debug
		}
		else_if = {
			limit = {
				has_global_variable = com_debug
			}
			remove_global_variable = com_debug
		}
	}
}

com_toggle_debug_sgui = {
	scope = none
	effect = {
		if = {
			limit = {
				NOT = {
					has_global_variable = com_toggle_debug_mode
				}
			}
			set_global_variable = com_toggle_debug_mode
		}
		else = {
			remove_global_variable = com_toggle_debug_mode
		}
	}
}
