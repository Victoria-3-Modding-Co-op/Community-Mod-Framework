}

macro = {
    description = "Get progress bar drift script value"
    definition = "ComProgressBarDriftSValue"

macro = {
    description = "Is progress bar value increasing"
    definition = "ComProgressBarIsIncreasing"
    replace_with = "GreaterThan_float(FixedPointToFloat(ComProgressBarDriftSValue), JournalEntry.GetCurrentBarProgress(ScriptedProgressBar.Self) )"
}

macro = {
    description = "Is progress bar value decreasing"
    definition = "ComProgressBarIsDecreasing"
    replace_with = "GreaterThan_float(JournalEntry.GetCurrentBarProgress(ScriptedProgressBar.Self), FixedPointToFloat(ComProgressBarDriftSValue))"
}
